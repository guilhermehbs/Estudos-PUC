CREATE TABLE CLIENTE (
    codigo_cliente int PRIMARY KEY,
    nome varchar,
    contato varchar,
    endereco varchar,
    bairro varchar,
    cidade varchar,
    estado varchar,
    cpf varchar,
    data_nascimento date
);

CREATE TABLE ANIMAL_VENDA_SERVICOS (
    codigo_animal int,
    especie varchar,
    valor double,
    caminho_imagem varchar,
    descricao varchar,
    FK_CLIENTE_codigo_cliente int,
    FK_FORNECEDOR_codigo_fornecedor int,
    codigo_venda int,
    data date,
    quantidade int,
    forma_pagamento varchar,
    codigo_servico int,
    tempo_servico varchar,
    servico_realizado varchar,
    PRIMARY KEY (codigo_animal, codigo_venda, codigo_servico)
);

CREATE TABLE NOTA_FISCAL (
    codigo_nota int PRIMARY KEY,
    FK_VENDA_codigo_venda int
);

CREATE TABLE FUNCIONARIO (
    codigo_funcionario int PRIMARY KEY,
    nome varchar,
    funcao varchar,
    cpf varchar
);

CREATE TABLE PRODUTOS (
    codigo_produto int PRIMARY KEY,
    nome varchar,
    valor double,
    categoria varchar,
    descricao varchar,
    caminho_imagem varchar,
    FK_FORNECEDOR_codigo_fornecedor int,
    FK_VENDA_codigo_venda int
);

CREATE TABLE FORNECEDOR (
